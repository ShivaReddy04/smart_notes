# requirements.txt
# Why this file exists:
#   Single source of truth for the project's Python dependencies so the
#   environment is reproducible across machines and deployment targets.
#   Versions are pinned with a lower bound to balance stability with
#   security patches. We only list what Phase 1 actually needs; later
#   phases (LangChain, pgvector, etc.) add their own dependencies.

# --- Web framework ---
fastapi>=0.115.0          # ASGI web framework for the REST API
uvicorn[standard]>=0.32.0 # ASGI server to run FastAPI in dev and prod
python-multipart>=0.0.9   # Parses multipart/form-data (Phase 5 image uploads)

# --- Configuration & validation ---
pydantic>=2.9.0           # Data validation for request/response schemas
pydantic-settings>=2.6.0  # Loads & validates settings from environment/.env

# --- Database ---
sqlalchemy>=2.0.0         # ORM for modeling and querying PostgreSQL
psycopg2-binary>=2.9.0    # PostgreSQL driver used by SQLAlchemy
alembic>=1.13.0           # Database schema migrations (versioned)

# --- AI / LLM (Phase 2) ---
# We talk to OpenRouter (an OpenAI-compatible gateway to many models) using
# LangChain's primitives. We deliberately avoid the heavy umbrella
# `langchain` package and depend only on what we use.
langchain-core>=0.3.0     # Prompt templates, output parsers, Runnable pipes
langchain-openai>=0.2.0   # ChatOpenAI client; points at OpenRouter's base_url

# --- Embeddings & vector search (Phase 3) ---
# Local, on-device embeddings + a local persistent vector store. No API
# cost and full privacy, at the price of a larger install.
# NOTE: sentence-transformers depends on PyTorch (torch), which is a large
# download (hundreds of MB). The embedding model weights (~80MB for
# all-MiniLM-L6-v2) are fetched once on first use and cached locally.
sentence-transformers>=3.0.0  # all-MiniLM-L6-v2 text embeddings
chromadb>=0.5.0               # Local persistent vector database
